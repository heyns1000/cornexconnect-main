            const activeLink = document.getElementById(activeId);
            if (activeLink) {
                activeLink.classList.remove('hover:bg-gray-700');

        // --- 4. SPA Router Logic (Now primarily for Sector Dashboard and "Coming Soon" brands) ---
        function handleRouting() {
            const path = window.location.hash.slice(1);
            if (path.startsWith('brand-')) {
                const brandName = path.substring(6);
                renderBrandDashboard(brandName); // This will now render the "Coming Soon" page for uncompleted brands
            } else if (path === 'sector-dashboard' || path === '') {
                renderSectorDashboard();
            } else {
                // For any other hash, or direct access to root, go to sector dashboard
                window.location.hash = '#sector-dashboard';
                renderSectorDashboard();
            }
        }

            if (window.location.pathname === '/' && window.location.hash === '') {
                window.location.hash = '#sector-dashboard';
            }
            handleRouting();
        });
